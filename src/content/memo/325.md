function formatToManKm(distance: number): string {
// 10,000으로 나누고 소수점 한 자리까지 표현
const manKm = (distance / 10000).toFixed(1);
return `${manKm}만km`;
}

// 사용 예시
const formatted = formatToManKm(104335); // "10.4만km"
console.log(formatted);
